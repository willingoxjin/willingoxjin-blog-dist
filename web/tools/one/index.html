# 一、jsPDF

Github地址：https://github.com/parallax/jsPDF

# 1.安装

这里我是在 vue 项目中使用，通过 npm 工具安装

npm install jspdf --save

# 2.手动生成PDF

import {jsPDF as JsPDF} from 'jspdf'

// Default export is a4 paper, portrait, using millimeters for units
const doc = new JsPDF();

doc.text("Hello world!", 10, 10);
doc.save("a4.pdf");

# 3.页面DOM生成PDF

jsPDF 提供了转换 html 为 PDF 的API。如下是在 vue3 中使用：

<template>
    <table id="table" width="100%" border="1" cellspacing="0" cellpadding="0">
        <caption>
        	<h2>个人简历</h2>
        </caption>
        <tbody>
            <tr>
                <td><p>姓名</p></td>
                <td><p>王家虎</p></td>
                <td><p>性别</p></td>
                <td><p>男</p></td>
                <td><p>年龄</p></td>
                <td><p>22</p></td>
                <td>
                	<img src="@/assets/images/article-p1.jpg" alt="" style="width: 70px; height: 93px">
                </td>
            </tr>
        </tbody>
    </table>
    <div class="sub-info-opt">
      <a href="javascript:;" @click="downloadPDF()">下载</a>
    </div>
</template>
<script setup>
import {jsPDF as JsPDF} from 'jspdf'

/**
 * html转pdf
 * @param element Don对象
 * @param filename 生成的文件名
 */
function html2PDF(element, filename = new Date().getTime()) {
  const doc = new JsPDF('p', 'pt', 'a4')

  doc.setFont('PingFang-normal')

  doc.html(element, {
    filename: filename,
    margin: [50, 50, 50, 50],
    autoPaging: true,
    windowWidth: 695,
    width: 500,
    callback: (_pdf) => {
      _pdf.save(filename + '.pdf')
    },
  })
}

const downloadPDF = () => html2PDF(document.querySelector('#table'))

</script>

# 4.中文问题

不仅是中文，还有Unicode字符/ UTF-8，都需要引入字体文件。但是对于浏览器用户来说，访问时就需要下载下来字体，这就会有字体库过大导致下载很慢的问题。

你可以将字体库通过官方提供的工具：https://rawgit.com/MrRio/jsPDF/master/fontconverter/fontconverter.html 处理。通过该工具会得到一个 js文件，这个文件就是将字体库Base64后，通过js代码引入字体。

注意：在 html 中使用的字体，必须和我们导入的字体相同，这样导出的 PDF 字体才会生效。

然后将生成的 js 文件导入到代码中即可。示例代码如下：

import '@/assets/font/PingFang-normal.js' // 导入

import {jsPDF as JsPDF} from 'jspdf'

function html2PDF(element, filename = new Date().getTime()) {
  const doc = new JsPDF('p', 'pt', 'a4')
  // 设置字体
  doc.setFont('PingFang-normal') // 设置使用的字体

  doc.html(element, {
    filename: filename,
    margin: [50, 50, 50, 50],
    autoPaging: true,
    windowWidth: 695,
    width: 500,
    callback: (_pdf) => {
      _pdf.save(filename + '.pdf')
    },
  })
}

# 二、html2Canvas

html2Canvas 可以方便的将Dom转换为图片。

根据这个思路，为了解决中文问题，我们可以通过 html2Canvas 将 Dom 转换为图片，然后将图片放到 PDF 中。

示例代码如下：

<template>
    <table id="table" width="100%" border="1" cellspacing="0" cellpadding="0">
        <caption>
        	<h2>个人简历</h2>
        </caption>
        <tbody>
            <tr>
                <td><p>姓名</p></td>
                <td><p>王家虎</p></td>
                <td><p>性别</p></td>
                <td><p>男</p></td>
                <td><p>年龄</p></td>
                <td><p>22</p></td>
                <td>
                	<img src="@/assets/images/article-p1.jpg" alt="" style="width: 70px; height: 93px">
                </td>
            </tr>
        </tbody>
    </table>
    <div class="sub-info-opt">
      <a href="javascript:;" @click="downloadPDF()">下载</a>
    </div>
</template>

<script setup>
import html2Canvas from 'html2canvas'

/**
 * html转pdf
 * @param element Don对象
 * @param filename 生成的文件名
 */
function html2Canvas2CPDF(element, filename = new Date().getTime()) {
  // 当下载pdf时，若不在页面顶部会造成PDF样式不对,所以先回到页面顶部再下载
  let top = document.getElementById('pdfDom');
  if (top != null) {
    top.scrollIntoView();
    top = null;
  }
  html2Canvas(element, {
    allowTaint: true,
    margin: [50, 50, 50, 50],
    scale: 4,
  }).then(canvas => {
    // 获取canvas画布的宽高
    let contentWidth = canvas.width;
    let contentHeight = canvas.height;
    // 一页pdf显示html页面生成的canvas高度;
    let pageHeight = contentWidth / 841.89 * 592.28;
    // 未生成pdf的html页面高度
    let leftHeight = contentHeight;
    // 页面偏移
    let position = 0;
    // html页面生成的canvas在pdf中图片的宽高（本例为：横向a4纸[841.89,592.28]，纵向需调换尺寸）
    let imgWidth = 841.89;
    let imgHeight = 841.89 / contentWidth * contentHeight;
    let pageData = canvas.toDataURL('image/jpeg', 1.0);

    let PDF = new JsPDF('p', 'pt', 'a4')
    // 两个高度需要区分: 一个是html页面的实际高度，和生成pdf的页面高度
    // 当内容未超过pdf一页显示的范围，无需分页
    if (leftHeight < pageHeight) {
      PDF.addImage(pageData, 'JPEG', 0, 0, imgWidth, imgHeight)
    } else {
      while (leftHeight > 0) {
        PDF.addImage(pageData, 'JPEG', 0, position, imgWidth, imgHeight)
        leftHeight -= pageHeight;
        position -= 592.28;
        // 避免添加空白页
        if (leftHeight > 0) {
          PDF.addPage();
        }
      }
    }
    PDF.save(filename + '.pdf')
  })
}

const downloadPDF = () => html2Canvas2CPDF(document.querySelector('#table'))

</script>

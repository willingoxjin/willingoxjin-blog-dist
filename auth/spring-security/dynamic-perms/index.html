11.Spring Security 的动态权限控制

# 一、动态权限实现思路

# 1.动态权限

虽然我们可以通过代码来配置 HTTP 请求中的 URL 拦截规则和请求 URL 所需要的权限，或者使用相关的注解配置方法的权限。不过，一旦我们想要修改 HTTP 请求的 URL 对应权限，就需要修改代码，这样使用起来不是非常方便。

如果我们把 URL 拦截规则和对应的权限存储到数据库，在校检权限时，只需要从数据库查询对应的 URL 和对应的匹配规则，当我们修改了 URL 拦截规则和权限，在权限认证时，也能获取到最新的 URL 规则和权限。

# 2.实现思路

这里，我们基于 RDBC0 模型实现动态权限，如果修改了数据库，用户的权限也可以根据数据库动态的同步。

在前面的权限控制的学习中，我们了解到 Spring Security 可以为匹配到的请求配置对应的 AuthorizationManager。如下：

http
    .authorizeHttpRequests(auth -> auth
   			.requestMatchers("/resource/{name}").access(new WebExpressionAuthorizationManager("#name == authentication.name"))
        .anyRequest().authenticated()
    )

这样，我们就可以仿照 WebExpressionAuthorizationManager 或 AuthorizationManager 的其他实现类，自定义实现 AuthorizationManager ，在我们的实现类中我们自己的动态权限的认证了。并将所有的请求配置成使用我们自定义的 AuthorizationManager 进行权限校检就可以了。

简单示例如下：

public class DynamicAuthorizationManager implements AuthorizationManager<RequestAuthorizationContext> {

  	@Override
    public AuthorizationDecision check(Supplier<Authentication> authentication, RequestAuthorizationContext context) {
     		// TODO 根据Redis或其他存储查询规则,并校检用户的权限
      	return new AuthorizationDecision(true);
    }

}

# 2.2 UserDetailsService 实现

@Service
public class MybatisUserDetailsServiceImpl implements UserDetailsService, InitializingBean {

    private static final String IS_DISABLED = "0";

    private static String DEFAULT_ROLE_PREFIX = "ROLE_";

    @Autowired
    private SysUserMapper sysUserMapper;

    @Autowired
    private SysUserRoleMapper sysUserRoleMapper;

    @Autowired
    private SysRoleMenuMapper sysRoleMenuMapper;

    @Autowired
    private GrantedAuthorityDefaults grantedAuthorityDefaults;

    @Override
    public void afterPropertiesSet() throws Exception {
        DEFAULT_ROLE_PREFIX = grantedAuthorityDefaults.getRolePrefix();
    }

    @Override
    public UserDetails loadUserByUsername(String username) throws UsernameNotFoundException {
        // 查询用户信息
        SysUser user = sysUserMapper.selectOne(Wrappers.<SysUser>lambdaQuery().eq(SysUser::getUsername, username));
        if (user == null) {
            throw new UsernameNotFoundException(username);
        }

        String[] roles = this.getUserRoles(user.getUserId());
        UserBuilder userBuilder = User.withUsername(user.getUsername())
                .password(user.getPassword())
                .disabled(user.getEnable().equals(IS_DISABLED));
        if (roles != null) {
            userBuilder.authorities(roles);
        }

        return userBuilder.build();
    }

    private String[] getUserRoles(Long userId) {
        List<UserRoleVO> userRoleList = sysUserRoleMapper.selectUserRoleVOByUser(userId);
        if (CollectionUtils.isEmpty(userRoleList)) {
            return null;
        }
        Set<String> roleKeys = userRoleList.stream().map(UserRoleVO::getRoleKey)
                .collect(Collectors.toSet());
        List<AuthorityPatten> authorityPattens = sysRoleMenuMapper.selectPattenByRoleKeys(roleKeys);
        if (CollectionUtils.isEmpty(authorityPattens)) {
            return null;
        }

        return authorityPattens.stream().filter(Objects::nonNull)
                .flatMap(a -> {
                    String[] authorities = new String[] {
                            a.getPerms(),
                            DEFAULT_ROLE_PREFIX + a.getRole()
                    };

                    return Stream.of(authorities);
                })
                .filter(StringUtils::hasText)
                .distinct()
                .toArray(String[]::new);
    }

}

# 2.3 SysRoleMenuService 实现

@Service
public class SysRoleMenuService implements InitializingBean {

    private static String DEFAULT_ROLE_PREFIX = "ROLE_";

    @Autowired
    private SysRoleMenuMapper sysRoleMenuMapper;

    @Autowired
    private GrantedAuthorityDefaults grantedAuthorityDefaults;

    @Override
    public void afterPropertiesSet() throws Exception {
        DEFAULT_ROLE_PREFIX = grantedAuthorityDefaults.getRolePrefix();
    }

    public List<AuthorityPatten> getAllPatten() {
        List<AuthorityPatten> authorityPattens = sysRoleMenuMapper.selectAllPatten();
        this.setRolePrefix(authorityPattens);
        return authorityPattens;
    }

    public List<AuthorityPatten> getPattenByRoles(List<String> role) {
        List<AuthorityPatten> authorityPattens = sysRoleMenuMapper.selectPattenByRoleKeys(role);
        this.setRolePrefix(authorityPattens);

        return authorityPattens;
    }

    private void setRolePrefix(List<AuthorityPatten> authorityPattens) {
        if (CollectionUtils.isEmpty(authorityPattens)) {
            return;
        }
        for (AuthorityPatten authorityPatten : authorityPattens) {
            authorityPatten.setRole(DEFAULT_ROLE_PREFIX + authorityPatten.getRole());
        }
    }

}

# 2.4 权限规则查询接口

public interface AuthorizationPattenStore {

  	/**
     * 获取所有的请求匹配与权限规则
     * 
     * @return 所有的请求匹配与权限规则 {@link AuthorityPatten}
     */
    List<AuthorityPatten> getAllPatten();

    /**
     * 根据权限获取所有的请求匹配与权限规则
     *
     * @return 所有的请求匹配与权限规则 {@link AuthorityPatten}
     */
    List<AuthorityPatten> getPattens(Collection<String> authorities);

    default List<AuthorityPatten> getPattensByGrantedAuthority(Collection<? extends GrantedAuthority> authorities) {
        Set<String> set = authorities.stream().map(GrantedAuthority::getAuthority).collect(Collectors.toSet());
        return getPattens(set);
    }

}

# 2.5 AuthorizationManager 实现

public class DynamicAuthorizationManager implements AuthorizationManager<RequestAuthorizationContext> {

    private static final String ROLE_SUPER_ADMIN = "ROLE_SUPER_ADMIN";

    private static final String AUTHORITY_SUPER_ADMIN = "*:*:*";

    private static final String ALL_METHODS = "ALL";

    private AuthorizationPattenStore authorizationPattenStore;

    private PathMatcher antPathMatcher = new AntPathMatcher();

    private AuthoritiesAuthorizationManager delegate = new AuthoritiesAuthorizationManager();

    public DynamicAuthorizationManager(AuthorizationPattenStore authorizationPattenStore) {
        this.authorizationPattenStore = authorizationPattenStore;
    }

    @Override
    public AuthorizationDecision check(Supplier<Authentication> authentication, RequestAuthorizationContext context) {
        Authentication user = authentication.get();
        Collection<? extends GrantedAuthority> userAuthorities = user.getAuthorities();
        if (userAuthorities == null || userAuthorities.isEmpty()) {
            return new AuthorizationDecision(false);
        }
        // 校检是否为超管
        if (isSuperAdmin(userAuthorities)) {
            return new AuthorizationDecision(true);
        }
        // 匹配请求路径和方法
        List<AuthorityPatten> pattens = authorizationPattenStore.getPattensByGrantedAuthority(userAuthorities);
        if (matchPatten(context.getRequest(), pattens)) {
            // 校检是否有权限
            Set<String> authorities = pattens.stream().map(AuthorityPatten::getRole).collect(Collectors.toSet());
            return this.delegate.check(authentication, authorities);
        }

        return new AuthorizationDecision(false);
    }

    private boolean isSuperAdmin(Collection<? extends GrantedAuthority> authorities) {
        for (GrantedAuthority authority : authorities) {
            boolean isSuperAdmin = authority.getAuthority().equalsIgnoreCase(ROLE_SUPER_ADMIN)
                    || authority.getAuthority().equals(AUTHORITY_SUPER_ADMIN);
            if (isSuperAdmin) {
                return true;
            }
        }
        return false;
    }

    private boolean matchPatten(HttpServletRequest request, Collection<AuthorityPatten> pattens) {
        if (pattens == null || pattens.isEmpty()) {
            return false;
        }
        // 这里调用其他的AuthorizationManager进行权限校验
        for (AuthorityPatten patten : pattens) {
            // 转换为Matches对象
            boolean matchUrl = this.antPathMatcher.match(patten.getPatten(), request.getRequestURI());
            if (matchUrl && matchMethod(patten.getMethodSet(), request.getMethod())) {
                return true;
            }
        }
        return false;
    }


    private boolean matchMethod(Set<String> methods, String requestMethod) {
        if (methods == null || methods.isEmpty()) {
            return false;
        }
        for (String method : methods) {
            if (ALL_METHODS.equalsIgnoreCase(method) || method.equalsIgnoreCase(requestMethod)) {
                return true;
            }
        }
        return false;
    }

    public void setAuthorizationPattenStore(AuthorizationPattenStore authorizationPattenStore) {
        this.authorizationPattenStore = authorizationPattenStore;
    }

    public void setAntPathMatcher(PathMatcher antPathMatcher) {
        this.antPathMatcher = antPathMatcher;
    }

    public void setDelegate(AuthoritiesAuthorizationManager delegate) {
        this.delegate = delegate;
    }
}

@Configuration
public class SecurityConfig {

    @Autowired
    private AuthenticationManager authenticationManager;

    @Autowired
    private DynamicAuthorizationManager dynamicAuthorizationManager;

 		@Bean
    public SecurityFilterChain securityFilterChain(HttpSecurity http) throws Exception {
      
        http
            .authorizeHttpRequests(auth -> auth
                // 不校检OPTIONS方法
            		.requestMatchers(HttpMethod.OPTIONS).permitAll()
               	// 动态校检admin开头的请求
                .requestMatchers("/admin/**").access(dynamicAuthorizationManager)
                .anyRequest().authenticated()
            )
           	.formLogin(formLogin -> formLogin.permitAll()
           	.authenticationManager(authenticationManager)
            .csrf(AbstractHttpConfigurer::disable);

        return http.build();
    }

}

# 3.前端权限

对于前端来说，可以通过接口查询出当前登录用户所拥有的目录、菜单权限来动态渲染。

  • 如果当前登录用户没有某个目录、菜单或按钮的权限，那我们就不必显示在页面了。
  • 对于目录或者菜单，可以根据接口返回的当前用户所拥有的菜单数据，动态的渲染目录和菜单即可。

对于按钮来说，可以使用如下方式：

如果你使用的是 Vue，那么你可以使用自定义指令来判断是否显示按钮，步骤如下：

  1. 请求接口，获取当前登录用户的所有权限，将权限存储到 Vuex 中（也可以存储到 sessionStorage 或其他存储）；
  2. 自定义指令：v-hasPerm

自定义指令代码：

// src/directives/hasPerms.ts

import { Directive, DirectiveBinding } from 'vue'

// 超管角色
const superPermission = '*:*:*'

/**
 * 判断当前用户是否拥有权限
 */
const hasPerms = (permissions: Array<string>): boolean => {
  // TODO 这里应该从Vuex中获取用户角色
  const userPermissions = ['USER', 'ADMIN']
  return userPermissions.some((perm: string) => {
    return superPermission === role || permissions.includes(perm)
  })
}

/**
 * 自定义指令
 */
const hasPermsDirective: Directive = {
  mounted(el: HTMLElement, binding: DirectiveBinding<Array<string>>) {
    const { value } = binding
    if (!value || value.length === 0) {
      throw new Error('hasRole directive params error')
    }

    // 没有权限,则使按钮不显示
    if (!hasPerms(value)) {
      // el.style.display = 'none'
      el.parentNode && el.parentNode.removeChild(el)
    }
  }
}

export default hasPermDirective

# 4.其他

当然，上面的可以再优化一下，可以自定义一个 Redis 实现的 AuthorizationPattenStore，你可以命名为 AuthorizationPattenRedisStore ，初始化时这个对象时，把角色与资源的规则从数据库同步到 Redis 中。这样查询时性能相对更好。但是，如果使用 Redis 就需要在后台更新数据库时，同时更新 Redis 缓存的数据，我们可以通过 Spring 提供的 ApplicationEvent 来触发缓存更新。

这里我也给出了实现：

public class AuthorizationPattenRedisStore implements AuthorizationPattenStore, InitializingBean {

    private static final String AUTHORITIES_REDIS_KEY = "requestAuthorizationPatten";

    private AuthorizationPattenJdbcStore authorizationPattenJdbcStore;

    private RedisTemplate<String, Object> redisTemplate;

    private HashOperations<String, String, List<AuthorityPatten>> hashOperations;

    private AtomicBoolean init = new AtomicBoolean(false);

    public AuthorizationPattenRedisStore() {
    }

    public AuthorizationPattenRedisStore(AuthorizationPattenJdbcStore authorizationPattenJdbcStore,
            RedisTemplate<String, Object> redisTemplate) {
        this.authorizationPattenJdbcStore = authorizationPattenJdbcStore;
        this.redisTemplate = redisTemplate;
        this.initPattens();
    }

    @Override
    public void afterPropertiesSet() throws Exception {
        this.initPattens();
    }

    public void initPattens() {
        if (init.compareAndSet(false, true)) {
            Assert.notNull(authorizationPattenJdbcStore, "authorizationPattenJdbcStore must not be null");
            Assert.notNull(redisTemplate, "redisTemplate must not be null");
            hashOperations = redisTemplate.opsForHash();

            this.refreshPattens();
        }
    }

    /**
     * 若在管理端修改了相关的角色或权限，可以使用事件监听机制，
     * 调用此方法，重新加载权限到redis
     */
    public void refreshPattens() {
        // 若在管理端修改了相关的角色或权限，可以使用事件监听机制，重新加载权限到redis
        // 从数据库中查询角色对应的权限
        List<AuthorityPatten> allPatten = authorizationPattenJdbcStore.getAllPatten();
        Map<String, List<AuthorityPatten>> pattenMap;
        if (CollectionUtils.isEmpty(allPatten)) {
            pattenMap = new HashMap<>();
        } else {
            pattenMap = allPatten.stream().collect(Collectors.groupingBy(AuthorityPatten::getRole));
        }
        // 将权限存入redis
        // 删除redis中的所有权限
        redisTemplate.delete(AUTHORITIES_REDIS_KEY);

        hashOperations.putAll(AUTHORITIES_REDIS_KEY, pattenMap);
    }

    @Override
    public List<AuthorityPatten> getAllPatten() {
        List<List<AuthorityPatten>> cache = hashOperations.values(AUTHORITIES_REDIS_KEY);
        if (CollectionUtils.isEmpty(cache)) {
            return null;
        }
        return toPattenList(cache);
    }

    @Override
    public List<AuthorityPatten> getPattens(Collection<String> authorities) {
        List<List<AuthorityPatten>> cache = hashOperations.multiGet(AUTHORITIES_REDIS_KEY, List.copyOf(authorities));
        if (CollectionUtils.isEmpty(cache)) {
            return null;
        }
        return toPattenList(cache);
    }

    private List<AuthorityPatten> toPattenList(List<List<AuthorityPatten>> lists) {
        return lists.stream().filter(Objects::nonNull)
                .flatMap(Collection::stream)
                .collect(Collectors.toList());
    }

    public void setRedisTemplate(RedisTemplate<String, Object> redisTemplate) {
        this.redisTemplate = redisTemplate;
    }

    public void setAuthorizationPattenJdbcStore(AuthorizationPattenJdbcStore authorizationPattenJdbcStore) {
        this.authorizationPattenJdbcStore = authorizationPattenJdbcStore;
    }

}

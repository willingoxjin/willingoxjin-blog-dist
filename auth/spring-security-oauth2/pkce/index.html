09.Spring Security OAuth2 使用授权码增强模式（PKCE）

Spring Authorization Server–使用带有PKCE的单页应用程序进行身份验证 (opens new window)

# 一、客户端中使用授权码增强模式

# 1.相关源码

spring-securitry-oauth2–client：

  • OAuth2AuthorizationRequestCustomizers：用于自定义 OAuth2 授权请求。

    • 在该类的 applyPkce 方法中，对请求参数追加了 code_challenge、code_challenge_method 参数。
      • code_challenge_method 参数默认使用 S256 （即 SHA-256 算法）
      • code_challenge ：code_verifier 被 code_challenge_method 算法加密后的内容（code_verifier 是随机生成的 Base64 字符串）。

  • OAuth2AuthorizationCodeGrantRequestEntityConverter：用于获取授权码时，若使用 PKCE 模式，则会添加 code_verifier 参数。

spring-securitry-oauth2–authorization-server：

  • OAuth2AuthorizationCodeRequestAuthenticationProvider：该类在处理授权码请求时，会对``code_challenge、code_challenge_method` 参数进行校检。
  • CodeVerifierAuthenticator：用于验证 code_verifier 参数。

# 2.使用示例

我们在使用时，只需要修改客户端中 application.yml 中关于 spring.security.oauth2 中的配置，就会默认使用 PKCE 模式。

示例如下：

  1. 修改客户端 application.yml 的配置：
# spring security oath2 client
spring:
  security:
    oauth2:
      client:
        registration:
          mall:
            provider: mall
            client-name: mall
            client-id: mall-client-id
            client-secret: mall-client-secret
            # 授权方法设置为none，则会使用PKCE模式授权
            client-authentication-method: none
            # 使用授权码模式授权
            authorization-grant-type: authorization_code
            scope:
              - openid
              - profile
        provider:
          mall:
            issuer-uri: http://oauth2-server.local:21000

src/utils/crypto.js

import sha256 from 'crypto-js/sha256'
import base64url from 'crypto-js/enc-base64url.js'

/**
 * 生成随机字符串
 * @param length {number} 字符串长度
 * @return {string} 随机字符串
 */
export function generateRandomString(length = 32) {
  const possible = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789'
  let text = ''
  for (let i = 0; i < length; i++) {
    text += possible.charAt(Math.floor(Math.random() * possible.length))
  }
  return text
}

/**
 * 生成code_verifier
 */
export function generateCodeChallenge(code_verifier) {
  return base64url.stringify(sha256(code_verifier))
}

src/utils/request.js

export default function request({ url, method = 'get', data, params, headers }) {
    if (params) {
        const paramsArray = []
        Object.keys(params).forEach(key => paramsArray.push(`${key}=${params[key]}`))
        if (url.search(/\?/) === -1) {
            url += `?${paramsArray.join('&')}`
        } else {
            url += `&${paramsArray.join('&')}`
        }
    }
    
    let body
    if (headers['Content-Type'] === 'multipart/form-data') {
        const formData = new FormData()
        Object.keys(data).forEach(key => formData.append(key, data[key]))
        body = formData
    } else if (headers['Content-Type'] === 'application/x-www-form-urlencoded') {
        const formData = new URLSearchParams()
        Object.keys(data).forEach(key => formData.append(key, data[key]))
        body = formData
    } else {
        // Default to json
        headers['Content-Type'] = 'application/json'
        body = JSON.stringify(data)
    }
    
    // Execute the request
    return fetch(url, {
        method,
        headers: {
            ...headers
        },
        body: body,
    }).then(res => {
        if (res.status >= 200 && res.status < 300) {
            return res.json()
        } else {
            return Promise.reject(res)
        }
    }).catch(err => {
        return Promise.reject(err)
    })
}

src/api/oauth2.js

import request from "@/utils/request.js"

const issuer = import.meta.env.VITE_APP_AUTHORIZATION_SERVER_ISSUER
const ISSUER_CACHE_KEY = 'oauth2_issuer'

export function requestOAuth2Issuer(cache = true) {
  return request({
    url: issuer,
    method: "get"
  }).then(res => {
    if (cache) {
      localStorage.setItem(ISSUER_CACHE_KEY, JSON.stringify(res))
    }
    return res
  })
}

export async function requestOAuth2Token(data) {
  const issuerStr = localStorage.getItem(ISSUER_CACHE_KEY)
  let issuer = issuerStr ? JSON.parse(issuerStr) : await requestOAuth2Issuer(false)
  return request({
    url: issuer.token_endpoint,
    method: "post",
    params: data
  })
}

src/utils/oauth2-code.js

import { requestOAuth2Issuer } from "@/api/oauth2.js"
import { generateCodeChallenge, generateRandomString } from "@/utils/crypto.js"

const client_id = import.meta.env.VITE_APP_CLIENT_ID
const redirect_uri = import.meta.env.VITE_APP_CLIENT_REDIRECT_URI
const scopes = import.meta.env.VITE_APP_CLIENT_SCOPES
const code_challenge_method = "S256"

function redirectToOAuth2CodeUrl(url) {
  const state = generateRandomString()
  localStorage.setItem("state", state)
  const code_verifier = generateRandomString()
  localStorage.setItem("code_verifier", code_verifier)
  const code_challenge = generateCodeChallenge(code_verifier)
  const scope = scopes.split(",").join(" ")

  // 跳转到授权服务器
  const redirectOAuth2CodeUrl = `${url}?code_challenge_method=${code_challenge_method}&code_challenge=${code_challenge}`
      + `&client_id=${client_id}&redirect_uri=${encodeURI(redirect_uri)}&response_type=code&scope=${scope}&state=${state}`
  window.location.href = encodeURI(redirectOAuth2CodeUrl)
}

export function handleOAuth2CodeLogin() {
  requestOAuth2Issuer()
  .then(res => {
    const authorization_endpoint = res.authorization_endpoint
    redirectToOAuth2CodeUrl(authorization_endpoint)
  })
}

<script setup>
import { requestOAuth2Token } from "@/api/oauth2.js"
import { useRouter } from "vue-router"

const router = useRouter()
const client_id = import.meta.env.VITE_APP_CLIENT_ID
const redirect_uri = import.meta.env.VITE_APP_CLIENT_REDIRECT_URI

/**
 * 获取url参数
 * @param variable 参数名
 * @return {string} 参数值
 */
function getQueryVariable(variable) {
  const query = window.location.search.substring(1)
  const vars = query.split('&')
  for (let i = 0; i < vars.length; i++) {
    const pair = vars[i].split('=')
    if (pair[0] === variable) {
      return pair[1]
    }
  }
  return ''
}

/**
 * 校检state
 */
async function checkState() {
  const state = localStorage.getItem("state")
  const _state = getQueryVariable('state')
  return state === _state
}

/**
 * 请求token
 */
function getOAuth2Token() {
  requestOAuth2Token({
    client_id,
    grant_type: "authorization_code",
    redirect_uri,
    code: getQueryVariable('code'),
    code_verifier: localStorage.getItem("code_verifier"),
  }).then(res => {
    // TODO: 保存token
    localStorage.setItem("oauth2_token", JSON.stringify(res))
    router.push({ name: "Home" })
  }).catch(err => {
    // TODO 获取提示信息
    err.json().then(errRes => {
      if (confirm(errRes.error)) {
        router.push({ name: "Login" })
      }
    })
  })
}

function init() {
  checkState().then(res => {
    if (res) {
      getOAuth2Token()
    } else {
      alert("state不匹配")
    }
  })
}
init()
</script>

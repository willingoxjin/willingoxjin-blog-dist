ECMAScript

参考：

  • 阮一峰-ES6 入门教程 (opens new window)

本文并非全面讲解 ECMAScript，只是对一些特性做些讲解，或者在工作常用的一些语法做介绍，如果你想系统的学习，我推荐：阮一峰-ES6 入门教程 (opens new window)，里面有讲解和示例，非常详细。

ECMAScript 简称 ES , 你可把它看作为一套标准的语言，而 JavaScript 就是实施了这一套标准的语言

ECMAScript 和 JavaScript 的关系：前者是后者的规格，后者是前者的一种实现

当前主流的浏览器使用的是 ECMAScript 5

ECMAScript 6.0（以下简称 ES6）是 JavaScript 语言的下一代标准，现在的一些浏览器也逐步的去使用这一套标准，但是还不能使用ES6最新的全部特性，借助一些工具（Babel，JSPM）我们就可以使用ES6的特性

# 设置对象的 prototype Objetc.setPrototypeOf

Javascript 中的每个对象都有 prototype 属性，Javascript 中对象的 prototype 属性的解释是：返回对象类型原型的引用。

每一个构造函数都有一个属性叫做原型。这个属性非常有用：为一个特定类声明通用的变量或者函数。

你不需要显式地声明一个 prototype 属性，因为在每一个构造函数中都有它的存在

let breakfast = {
    getDrink() {
        return 'tea'
    }
}

let dinner = {
    getDrink() {
        return 'beer'
    }
}

let sunday = Object.create(breakfast)

console.log(sunday.getDrink())	// tea
console.log(Object.getPrototypeOf(sunday) === breakfast)	// true

// 设置 prototype
Object.setPrototypeOf(sunday, dinner)

console.log(sunday.getDrink())	// beer
console.log(Object.getPrototypeOf(sunday) === dinner) // true

# 迭代器 Iterators

用于迭代

  • next() 继续返回迭代器的下一个对象
  • {value: xxx, done: true/false} next() 返回的对象
    • value 当前只针对额值
    • done 是否完成迭代
  • Generators 生成迭代器
/**
 * 自定义迭代器
 * @param foods 数组类型，食物
 */
function chef(foods) {
    let i = 0
    
    return {
        next() {
            let done = (i>=foods.length)
            let value = !done ? foods[i++] : undefined 
            return {
                value,
            	done   
            }
        }
    }
}

let zhangsan = chef(['tomatoes', 'eggs'])
console.log(zhangsan.next())	// {value: 'tomatoes', done: false}
console.log(zhangsan.next())	// {value: 'eggs', done: false}
console.log(zhangsan.next())	// {value: undefined, done: true}

# Set

Set 是一些数据的集合，Set 中的数据不允许重复，你可以理解 Set 是不可重复的数组

Set 常用方法：

  • add 添加一个元素到Set
  • size 获取当前Set存储元素的长度
  • has 判断是否存在某个元素
  • delete 删除某个元素
  • forEach 循环遍历
  • clear 清空所有元素
let iceCream = new Set()
console.log(iceCream)	// Set(0) {size: 0}

iceCream = new Set('绿豆雪糕')
// 需要注意的是，Set构造犯法会将字符分割
console.log(iceCream)	// Set(4) {'绿', '豆', '雪', '糕'}

// 添加元素
iceCream.add('草莓圣代')
console.log(iceCream)	// Set(5) {'绿', '豆', '雪', '糕', '草莓圣代'}

// 添加元素重据
iceCream.add('草莓圣代')
console.log(iceCream)	// Set(5) {'绿', '豆', '雪', '糕', '草莓圣代'}

// 获取Set长度
console.log(iceCream.size)	// 5

// 是否包含某个元素
console.log(iceCream.has('雪'))	// true

// 删除元素
iceCream.delete('绿')
console.log(iceCream)	// Set(4) {'豆', '雪', '糕', '草莓圣代'}

// 遍历
iceCream.forEach((item) => {
    console.log(item)
})

// 清空所有元素
iceCream.clear()
console.log(iceCream)	// Set(0) {size: 0}

# Map

  • set 设置元素
  • size 获取 Map 中的元素的个数
  • get 根据元素的 key 获取 value 值
  • has 根据 key 判断是否存在某个元素
  • forEach 遍历 Map
  • clear 清空 Map 中的所有元素
let food = new Map()
console.log(food)	// Map(0) {size: 0}

let fruit = {}, 
    cook = function () {}, 
    dessert = '甜点'

food.set(fruit, 'apple')
food.set(cook, 'knife')
food.set(dessert, 'cake')
food.set('drink', 'tea')

console.log(food)	// Map(3) {{…} => 'apple', ƒ => 'knife', '甜点' => 'cake'}
console.log(food.size)	// 3
// 获取某个元素
console.log(food.get(fruit))	// apple
console.log(food.get(cook))		// knife
console.log(food.get(dessert))	// cake
console.log(food.get('drink'))	// tea

// 是否存在某个元素
console.log(food.has(fruit))	// true

// 遍历
food.forEach((value, key) => {
    console.log(key, value)
})

// 清空所有元素
food.clear()
